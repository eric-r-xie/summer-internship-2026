cgm_manifest <- read_tsv("data/wearable_blood_glucose.tsv", 
                         show_col_types = FALSE)

cohort_table <- participants %>%
  transmute(
    person_id,
    age,
    study_group,
    clinical_site,
    recommended_split,
    cgm_availability = wearable_blood_glucose,
    wearable_availability = wearable_activity_monitor,
    clinical_data_availability = clinical_data
  ) %>%
  left_join(
    cgm_manifest %>% select(person_id, 
                            average_cgm_glucose = average_glucose_level_mg_dl),
    by = "person_id"
  )

# Table 3: data availability summary
table3 <- tibble(
  category = c(
    "CGM data",
    "Wearable data",
    "Clinical data",
    "CGM plus wearable data",
    "CGM plus wearable plus clinical data"
  ),
  count = c(
    sum(participants$wearable_blood_glucose == TRUE),
    sum(participants$wearable_activity_monitor == TRUE),
    sum(participants$clinical_data == TRUE),
    sum(participants$wearable_blood_glucose == TRUE 
        & participants$wearable_activity_monitor == TRUE),
    sum(participants$wearable_blood_glucose == TRUE 
        & participants$wearable_activity_monitor == TRUE 
        & participants$clinical_data == TRUE)

)

2.4 Modality Availability Heatmap

library(tidyverse)

modality_long <- participants %>%
  select(person_id, cardiac_ecg, clinical_data, environment,
         retinal_flio, retinal_oct, retinal_octa, retinal_photography,
         wearable_activity_monitor, wearable_blood_glucose) %>%
  pivot_longer(-person_id, names_to = "modality", values_to = "available")

ggplot(modality_long, aes(x = modality, y = factor(person_id), fill = available)) +
  geom_tile(color = "white", linewidth = 0.1) +
  scale_fill_manual(values = c("TRUE" = "steelblue", "FALSE" = "#f0f0f0")) +
  labs(title = "Figure 1: Modality Availability Heatmap",
       x = "Modality", y = "Participant", fill = "Available") +
  theme_minimal() +
  theme(axis.text.y = element_blank(),   # 100 IDs won't fit legibly on y-axis
        axis.text.x = element_text(angle = 45, hjust = 1))

When visualized, the participants’ data is complete across all modalities in this subset. There is no missing modalities to present or analyze.
