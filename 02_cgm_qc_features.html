Generalized Function to Test on Five Participants

process_one_participant <- function(json_path) {
  
  # --- Extract raw JSON ---
  raw <- fromJSON(json_path, simplifyVector = FALSE)
  person_id_val   <- raw$header$patient_id
  timezone_abbrev <- raw$header$timezone
  
  cgm <- map_dfr(raw$body$cgm, function(r) {
    tibble(
      person_id              = person_id_val,
      event_type             = r$event_type,
      time_utc               = r$effective_time_frame$time_interval$start_date_time,
      timezone_abbreviation   = timezone_abbrev,
      glucose_value_original  = as.character(r$blood_glucose$value),   # force consistent type
      glucose_value_numeric   = suppressWarnings(as.numeric(r$blood_glucose$value)),
      glucose_unit             = r$blood_glucose$unit
    )
  })
  
  # --- Step 1: EGV filter ---
  cgm <- cgm %>% filter(event_type == "EGV")
  
  # --- Step 2: parse UTC, clean ID ---
  cgm <- cgm %>%
    mutate(
      time_utc          = ymd_hms(time_utc, tz = "UTC"),
      person_id_numeric = as.numeric(str_remove(person_id, "AIREADI-"))
    )
  
  # --- Step 3: timezone / site consistency ---
  cgm <- cgm %>%
    left_join(tz_map, by = "timezone_abbreviation") %>%
    left_join(participants_mini %>% select(person_id, clinical_site),
              by = c("person_id_numeric" = "person_id")) %>%
    left_join(site_region_map, by = "clinical_site") %>%
    mutate(
      tz_site_consistent = expected_region == site_region,
      tz_flag_for_review  = is.na(iana_zone) | is.na(site_region) | !tz_site_consistent
    )
  
  # --- Step 4: local time ---
  participant_tz <- unique(cgm$iana_zone)
  stopifnot(length(participant_tz) == 1)
  
  cgm <- cgm %>%
    mutate(
      time_local = with_tz(time_utc, tzone = participant_tz),
      date_local = as_date(time_local),
      hour_local = hour(time_local),
      is_dst     = dst(time_local)
    )
  
  # --- Step 6: missing values ---
  n_missing_time    <- sum(is.na(cgm$time_utc))
  n_missing_glucose <- sum(is.na(cgm$glucose_value_original))
  
  # --- Step 7: duplicates ---
  n_raw <- nrow(cgm)
  cgm_clean <- cgm %>% distinct(person_id, time_utc, .keep_all = TRUE)
  n_duplicates <- n_raw - nrow(cgm_clean)
  
  # --- Step 8: unit check ---
  cgm_clean <- cgm_clean %>%
    mutate(unit_flag = glucose_unit != "mg/dL" | is.na(glucose_unit))
  
  # --- Step 9: unusual values ---
  cgm_clean <- cgm_clean %>%
    mutate(unusual_value_flag = glucose_value_numeric < 40 | glucose_value_numeric > 400)
  n_unusual <- sum(cgm_clean$unusual_value_flag, na.rm = TRUE)
  
  cgm_valid <- cgm_clean %>% filter(!unusual_value_flag)
  
  # --- Step 10: final reading-level table ---
  reading_level <- cgm_valid %>%
    mutate(person_id = person_id_numeric) %>%
    select(person_id, event_type, time_utc, timezone_abbreviation,
           time_local, date_local, hour_local, is_dst,
           glucose_value_original, glucose_value_numeric, glucose_unit)
  
  # --- Step 11: valid-day counts ---
  day_counts <- reading_level %>%
    count(person_id, date_local, name = "n_readings") %>%
    mutate(valid_day = n_readings >= 202)
  
  # --- Step 12: DST-crossing flag ---
  dst_crossing <- reading_level %>%
    group_by(person_id, date_local) %>%
    summarize(n_distinct_dst = n_distinct(is_dst), .groups = "drop") %>%
    filter(n_distinct_dst > 1)
  
  # --- return everything needed downstream, bundled in a list ---
  list(
    reading_level     = reading_level,
    day_counts        = day_counts,
    dst_crossing_days = dst_crossing,
    qc_summary = tibble(
      person_id          = unique(reading_level$person_id),
      raw_egv_readings    = n_raw,
      missing_timestamps  = n_missing_time,
      missing_glucose        = n_missing_glucose,
      duplicate_timestamps   = n_duplicates,
      unusual_values            = n_unusual,
      cleaned_readings          = nrow(reading_level),
      valid_days                = sum(day_counts$valid_day)
    )
  )
}
